 configgitgit --version = Verifica a versão do git
git init = Inicia uma repositorio vazio na pasta
git config --global user.name "username" = Cria usuario do git
git config --global user.email "email" = Cria email do git
git status = Verifica tudo que não foi salvo
git add . = Colocar arquivos na area de transferencia
git commit -m "Mensagem para identificar o commit"
git log = Mostra todos o log dos commits realizados
git remote add origin linkdorepositorio = Vincula a pasta local ao repositorio do git
git push -u origin master = fazer commit pela primeira vez
